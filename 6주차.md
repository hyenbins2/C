# 6주차
- 6주차 실습
- 6주차 복습
  1. while 문
  2. do while 문

1. 
```
#include <stdio.h>

int main() {
  int n, sum = 0, i = 1;
  printf("정수를 입력하시오:");
  scanf("%d", &n);

  while (i <= n) {
    sum += i;
    i++;
  }

  printf("1부터 %d까지의 합은 %d입니다.", n, sum);
  return 0;
}
```

```
#include <stdio.h>

int main(void) {
  int n, sum = 0, i = 1;
  printf("정수를 입력하시오: ");
  scanf("%d", &n);

  while (i <= n) {
    if (i % 2 == 0) {
      sum += i;
    }
    i++;
  }
  printf("1부터 %d까지의 짝수합은 %d입니다.", n, sum);
}
```

```
#include <stdio.h>

int main(void) {
  int n, sum = 0, i = 0;

  while (i < 5) {
    printf("값을 입력하시오:");
    scanf("%d", &n);
    sum += n;
    i++;
  }
  printf("합계는 %d입니다.", sum);
}


2.

```
#include <stdio.h>
int main(void) {
  int i = 0;
  do {
    printf("1---새로만들기\n");
    printf("2---파일열기\n");
    printf("3---파일닫기\n");
    printf("하나를 선택하시오:");
    scanf("%d", &i);
  } while (i < 1 || i > 3);
  printf("선택된 메뉴 = %d\n", i);
  return 0;
}
