{
    int number;

    printf("정수를 입력하시오:");
    scanf("%d", &number);

    if (number%2 == 0)
        printf("입력된 정수는 짝수입니다.\n");
    else
        printf("입력된 정수는 홀수입니다.\n");

    return 0;
}

#include <stdio.h>
int main(void)
{
    char ch;
    printf("문자를 입력하시오:");

    ch = getchar();
    if( ch >='A' && ch <= 'Z')
        printf("%c는 대문자입니다.\n", ch);
    else if (ch >= 'a' && ch <= 'z')
        printf("%c는 소문자입니다.\n", ch);
    else if (ch >= '0' && ch <= '9')
        printf("%c는 숫자입니다.\n", ch);
    else
        printf("%c는 기타문자입니다.\n", ch);
        
        return 0;
